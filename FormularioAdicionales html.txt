    <script>
      const initial = <?!= initialData ?>;
      const optOri = ["2049", "4071", "9001", "8059", "EFECTIVO", "YAPE", "ALMACÉN"];
      const optProv = ["TERCEROS", "PLANILLA HACE"];

      window.onload = () => {
        if(initial.materiales.length) initial.materiales.forEach(r => addRowMat(r)); else addRowMat();
        if(initial.manoObra.length) initial.manoObra.forEach(r => addRowDet('bodyMO', r)); else addRowDet('bodyMO');
        if(initial.movilidad.length) initial.movilidad.forEach(r => addRowDet('bodyMov', r)); else addRowDet('bodyMov');
      };

      function switchTab(contentId, btnId) {
        document.querySelectorAll('.content').forEach(c => c.classList.remove('active'));
        document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
        document.getElementById(contentId).classList.add('active');
        document.getElementById(btnId).classList.add('active');
      }

      // --- MATERIALES ---
      function addRowMat(v=[]) {
        const b = document.getElementById('bodyMat'); const tr = document.createElement('tr');
        
        let s = `<select onchange="validateRow(this)"><option value="" disabled ${!v[8]?'selected':''}>Origen</option>`;
        optOri.forEach(o => s += `<option value="${o}" ${v[8]==o?'selected':''}>${o}</option>`);
        
        const fechaVal = v[9] || initial.hoy;

        // Se agregan validaciones oninput para Descripción, Unidad, y los pares Plan/Real
        tr.innerHTML = `
          <td><input type="text" value="${v[0]||''}" oninput="validateRow(this)"></td>
          <td><input type="text" value="${v[1]||''}" oninput="validateRow(this)"></td>
          
          <td><input type="number" value="${v[2]||''}" oninput="calc(this); validateRow(this)"></td>
          <td><input type="number" value="${v[3]||''}" oninput="calc(this); validateRow(this)"></td>
          <td><input type="text" readonly value="${v[4]||''}"></td>
          
          <td><input type="number" value="${v[5]||''}" oninput="calc(this); validateRow(this)"></td>
          <td><input type="number" value="${v[6]||''}" oninput="calc(this); validateRow(this)"></td>
          <td><input type="text" readonly value="${v[7]||''}"></td>
          
          <td>${s}</select></td>
          <td><input type="date" value="${fechaVal}"></td>
          <td><button class="btn-del" onclick="this.closest('tr').remove()">🗑</button></td>`;
        b.appendChild(tr);
        validateRow(tr.querySelector('select')); // Validar estado inicial
      }

      // --- DETALLES (MO / MOV) ---
      function addRowDet(tid, v=[]) {
        const b = document.getElementById(tid); const tr = document.createElement('tr');
        let sp = `<select><option value="" disabled ${!v[0]?'selected':''}>Prov.</option>`;
        optProv.forEach(p => sp += `<option value="${p}" ${v[0]==p?'selected':''}>${p}</option>`);
        
        let so = `<select onchange="validateRow(this)"><option value="" disabled ${!v[4]?'selected':''}>Origen</option>`;
        optOri.filter(x=>x!=='ALMACÉN').forEach(o => so += `<option value="${o}" ${v[4]==o?'selected':''}>${o}</option>`);
        
        const fechaVal = v[5] || initial.hoy;

        tr.innerHTML = `<td>${sp}</select></td><td><input type="text" value="${v[1]||''}"></td>
          <td><input type="number" value="${v[2]||''}"></td>
          <td><input type="number" value="${v[3]||''}" oninput="validateRow(this)"></td>
          <td>${so}</select></td>
          <td><input type="date" value="${fechaVal}"></td>
          <td><button class="btn-del" onclick="this.closest('tr').remove()">🗑</button></td>`;
        b.appendChild(tr);
        validateRow(tr.querySelectorAll('select')[1]); 
      }

      function calc(el) {
        const r = el.closest('tr').querySelectorAll('input');
        r[4].value = ((parseFloat(r[2].value)||0)*(parseFloat(r[3].value)||0)).toFixed(2);
        r[7].value = ((parseFloat(r[5].value)||0)*(parseFloat(r[6].value)||0)).toFixed(2);
      }

      /**
       * VALIDACIÓN DE FILA 
       */
      function validateRow(el) {
        if (!el) return;
        const tr = el.closest('tr');
        if (!tr) return;
        
        const inputs = tr.querySelectorAll('input');
        const selects = tr.querySelectorAll('select');
        const tbodyId = tr.closest('tbody').id;

        let realVal = 0;
        let origenSelect;

        if (tbodyId === 'bodyMat') {
            // --- LÓGICA ESPECÍFICA PARA MATERIALES ---
            
            // 1. Descripción y Unidad Obligatorios
            const desc = inputs[0];
            const und = inputs[1];
            
            if (desc.value.trim() === "") desc.classList.add('invalid-field');
            else desc.classList.remove('invalid-field');

            if (und.value.trim() === "") und.classList.add('invalid-field');
            else und.classList.remove('invalid-field');

            // 2. Validación cruzada de PLANIFICADO (Cant vs PU)
            // Inputs: [2] Plan Cant, [3] Plan PU
            const planCant = parseFloat(inputs[2].value) || 0;
            const planPu = parseFloat(inputs[3].value) || 0;
            const planCantStr = inputs[2].value;
            const planPuStr = inputs[3].value;

            // Si uno tiene texto pero el otro es 0/vacío -> error
            // Ojo: usamos los strings para detectar si escribieron algo aunque no sea número válido aún
            if (planCantStr && !planPuStr) {
               inputs[3].classList.add('invalid-field'); 
            } else if (!planCantStr && planPuStr) {
               inputs[2].classList.add('invalid-field');
            } else {
               inputs[2].classList.remove('invalid-field');
               inputs[3].classList.remove('invalid-field');
            }

            // 3. Validación cruzada de REAL (Cant vs PU)
            // Inputs: [5] Real Cant, [6] Real PU
            const realCant = parseFloat(inputs[5].value) || 0;
            const realPu = parseFloat(inputs[6].value) || 0;
            const realCantStr = inputs[5].value;
            const realPuStr = inputs[6].value;

            if (realCantStr && !realPuStr) {
               inputs[6].classList.add('invalid-field');
            } else if (!realCantStr && realPuStr) {
               inputs[5].classList.add('invalid-field');
            } else {
               inputs[5].classList.remove('invalid-field');
               inputs[6].classList.remove('invalid-field');
            }

            // Preparar valores para la lógica de Origen
            // Inputs[7] es Real Total calculado
            realVal = parseFloat(inputs[7].value) || 0;
            origenSelect = selects[0];

        } else {
            // --- LÓGICA PARA MANO DE OBRA / MOVILIDAD ---
            realVal = parseFloat(inputs[2].value) || 0;
            origenSelect = selects[1];
        }

        // --- LÓGICA COMÚN: ORIGEN OBLIGATORIO SI HAY MONTO REAL ---
        if (realVal > 0 && !origenSelect.value) {
            origenSelect.classList.add('invalid-field');
        } else {
            origenSelect.classList.remove('invalid-field');
        }
      }

      // Validar todas las filas de una tabla antes de guardar
      function validateTable(bodyId) {
        const rows = document.querySelectorAll('#' + bodyId + ' tr');
        let isValid = true;
        rows.forEach(tr => {
            // Disparar validación
            const el = tr.querySelector('select'); // Usar cualquier elemento para invocar
            validateRow(el);
            // Chequeamos si hay rojos
            const invalidElements = tr.querySelectorAll('.invalid-field');
            if (invalidElements.length > 0) isValid = false;
        });
        return isValid;
      }

      function getTableData(id) {
        return Array.from(document.querySelectorAll('#'+id+' tr')).map(tr => Array.from(tr.querySelectorAll('input, select')).map(i => i.value)).filter(r => r[0] || r[1]);
      }

      function saveLocal(type) {
        let bodyId = '';
        if (type === 'MATERIALES') bodyId = 'bodyMat';
        else if (type === 'MANO_OBRA') bodyId = 'bodyMO';
        else if (type === 'MOVILIDAD') bodyId = 'bodyMov';

        if (!validateTable(bodyId)) {
            alert("⚠️ NO SE PUEDE GUARDAR\n\nRevise los campos marcados en rojo:\n- Descripción y Unidad son obligatorios (Materiales).\n- Si llena Cantidad, P.U. es obligatorio (y viceversa).\n- Si hay Monto Real, Origen es obligatorio.");
            return;
        }

        const st = document.getElementById('global-status'); st.innerText = "⏳ Guardando...";
        let data, func;
        if(type==='MATERIALES') { data = getTableData('bodyMat'); func = 'guardarSoloMateriales'; }
        if(type==='MANO_OBRA') { data = getTableData('bodyMO'); func = 'guardarSoloManoObra'; }
        if(type==='MOVILIDAD') { data = getTableData('bodyMov'); func = 'guardarSoloMovilidad'; }

        google.script.run.withSuccessHandler(r => { st.innerText = r.message; setTimeout(()=>st.innerText="",3000); }).withFailureHandler(e => { alert(e); })[func](data);
      }

      function saveAll() {
        const vMat = validateTable('bodyMat');
        const vMO = validateTable('bodyMO');
        const vMov = validateTable('bodyMov');

        if (!vMat || !vMO || !vMov) {
             alert("⚠️ NO SE PUEDE GUARDAR TODO\n\nHay errores de validación en las filas (campos rojos).");
             return;
        }

        const payload = { materiales: getTableData('bodyMat'), manoObra: getTableData('bodyMO'), movilidad: getTableData('bodyMov') };
        const st = document.getElementById('global-status'); st.innerText = "⏳ Guardando TODO...";
        google.script.run.withSuccessHandler(r => { st.innerText = r.message; setTimeout(()=>st.innerText="",3000); }).guardarAdicionalesTodo(payload);
      }
